pipeline {
    agent any

    environment {
        VENV_DIR = "venv"
        APP_DIR = "C:\\flask_app" // Change this to your deployment directory
    }

    stages {
        stage('Clone Repository') {
            steps {
                echo "Cloning the latest code from Git..."
                bat "git clone https://github.com/your-repo/flask-app.git %APP_DIR%"
            }
        }

        stage('Setup Environment') {
            steps {
                echo "Setting up Python virtual environment..."
                bat """
                cd %APP_DIR%
                python -m venv %VENV_DIR%
                call %VENV_DIR%\\Scripts\\activate
                pip install --upgrade pip
                pip install -r requirements.txt
                """
            }
        }

        stage('Run Tests') {
            steps {
                echo "Running unit tests..."
                bat """
                cd %APP_DIR%
                call %VENV_DIR%\\Scripts\\activate
                pytest --junitxml=test-results.xml
                """
            }
        }

        stage('Deploy Application') {
            steps {
                echo "Deploying Flask application..."
                bat """
                taskkill /F /IM gunicorn.exe || echo "No existing process found"
                cd %APP_DIR%
                call %VENV_DIR%\\Scripts\\activate
                start /B gunicorn --bind 0.0.0.0:8000 app:app
                """
            }
        }
    }

    post {
        success {
            echo "Deployment Successful!"
        }
        failure {
            echo "Deployment Failed!"
        }
    }
}
